登录

url: /RbSystem/login.do

输入 String:id, String:password
返回 Json{
	String:success("success": 成功； "failure": 失败),
	Data:{
		int:level(1:报销人员；2：审核人员；3：审核负责人；4：收单员；5：系统管理员),
		# 报销人员
		User:{
			String:name(用户姓名), 
			String:id(工号), 
			int:gender(1：男；2：女)，
			String:type,
			String:idCard(身份证号),
			String:register_time(注册时间),
			String:telphone(手机号码),
		}，
		Department：{
			String:id(部门编号),
			String:name(部门名称)
		}
		# 报销人员 End
		# 管理人员
		Admin:{
			String: id,
			int: level
		}
		# 管理人员 End
	}
}

首页获取首页信息

url: /RbSystem/homepage.do
输入 无
返回 Json{
	String:success("success": 成功； "failure": 失败),
	Data:{
		int:rb_state(0:没有报销单申请记录; 1:未提交；2:待审核; 3:审核中; 4:审核通过; 5:审核未通过; 6:待报销; 7:已完成)
		String:rb_id(用于新增报销单申请表页面加载时输入，这样后台不用重新查找报销单id)
	}
}

报销单申请界面获取报销单内容
url: /RbSystem/user/getRbForm.do
输入 String:rb_id, rb_state		//按表单传送
返回 Json{
	String:success("success": 成功； "failure": 失败),
	Data:{
		String:rb_id,
		int:rb_state(1:未提交；2:待审核; 3:审核中; 4:审核通过; 5:审核未通过; 6:待报销; 7:已完成)
		String:hospital,(可能为空)
		Datetime:s_time(上次编辑时间),
		referral:{
			(这是转诊单，可能为空)
			String:id,
			Date:date,
			String:pic
		},
		ghf[]:{
			(这是挂号费列表，可能为空)
			String:id,
			String:department(科室),
			Date:date(日期),
			int:cost(总金额),
			int:self_paid(自付金额),
			String:note(审核说明),
			String:pic
		},
		yymx[]:{
			(这是用药明细列表，可能为空)
			String:id,
			Date:date,
			int:cost,
			int:special_paid,
			int:part_paid,
			int:self_paid,
			String:note,
			String:detailed_pic,
			String:pspt_pic
		},
		wssm:{
			(这是外伤说明，可能为空)
			String:id,
			String:note,(这是报销人员自己添加的说明)
			String:stamp_pic,
			String:special_pic
		}
	}
}

用户点击选择图片完成后就将图片上传到服务器，服务器会返回该图片存储路径 
url: /RbSystem/upload.do
输入: file
返回: Json {
	String:success("success": 成功； "failure": 失败),
	Data:{
		String:path(该图片访问路径)
	}
}
