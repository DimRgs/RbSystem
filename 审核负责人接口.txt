获取报销单表格数据及二维码
url: /RbSystem/user/getQR.do
输入: int:rb_id
返回: Json{
	String:success,
	Data:{
		int:rb_id,
		String:QRCode,
		String:u_id(报销人员编号),
		String:u_name(报销人员姓名),
		int:u_type(报销人员类别),
		String:u_department(报销人员部门),
		timestamp:c_time,
		String:hospital,
		int:total_cost(总申报金额),
		int:total_self_paid(总自费金额),
		ghf:{
			total_cost(总金额),
			self_pro(自付比例),
			self_paid(总自付金额),
			count(凭证数量),
		}
		yymx:{
			total_cost,
			self_pro,
			self_paid,
			count
		}
	}
}

审核负责人 接口 

1.查找审核人员列表
url: /RbSystem/admin/getEList.do
输入: 无
返回: Json{
	String:success,
	Data:{
		int:totalNum,
		int:totalPage,
		EList[]:{
			String:id,
			String:password,
			int:level
		}
	}
}

2.修改审核人员信息
url: /RbSystem/admin/updateAdmin.do
输入：admin{
	String:id,
	String:opwd,
	String:npwd,
}
返回: Json{
	String: success
}

3.增加审核人员
url: /RbSystem/admin/addAdmin.do
输入：admin{
	String:id,
	String:pwd,
}
返回: Json{
	String: success
}
